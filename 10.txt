import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

from sklearn.datasets import load_breast_cancer
from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans

x = StandardScaler().fit_transform(load_breast_cancer().data)
c = KMeans(n_clusters=2, random_state=42).fit_predict(x)
p = PCA(2).fit_transform(x)

df = pd.DataFrame(p, columns=["pca1", "pca2"])
df["cluster"]=c
sns.scatterplot(data=df, x="pca1", y="pca2", hue="cluster")
plt.show()
