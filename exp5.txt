import numpy as np
import matplotlib.pyplot as plt
from collections import Counter

data = np.random.rand(100)

labels = ["class1" if x<=0.5 else "class2" for x in data[:50]]

def eculidean_distance(x1,x2):
    return abs(x1-x2)
def knn_classifier(train_data, test_point, train_labels, k):
    distances = [(eculidean_distance(test_point, train_data[i]),train_labels[i]) for i in range(len(train_data))]

    distances.sort(key = lambda x: x[0])
    k_nearest_neighbour = distances[:k]
    k_nearest_labels = [label for _, label in k_nearest_neighbour]
    return Counter(k_nearest_labels).most_common(1)[0][0]

train_data = data[:50]
train_labels = labels
test_data = data[50:]

k_values = [1,2,3,4,5,20,30]

for k in k_values:
    print(f"resutlts for k = {k}:")
    predictions =[]

    for i, test_point in enumerate(test_data, start =51):
        label = knn_classifier(train_data, test_point, train_labels, k)
        predictions.append(label)
        print(f"Point x{i} (value: {test_point:.4f}) is classified as {label}")
    print("\n")

    class1_points = [test_data[i] for i in range(len(test_data)) if predictions[i] == "class1"]
    class2_points = [test_data[i] for i in range(len(test_data)) if predictions[i] == "class2"]

    plt.figure()

    #train
    plt.scatter(train_data, [0]*len(train_data), c=["blue" if l=="class1" else "red" for l in train_labels], label = "Training data")

    #test
    plt.scatter(class1_points, [1]*len(class1_points), c="blue", marker="x", label ="class1 (test)")
    plt.scatter(class2_points, [1]*len(class2_points), c="red", marker="x", label = "class2 (test)")

    plt.title("knn results")
    plt.xlabel("data values")
    plt.ylabel("level")
    plt.legend()
    plt.grid()
    plt.show()
